import Link from 'next/link'
import { useEffect, useState } from 'react'

export function StatusIndicator() {
  const [status, setStatus] = useState({ text: 'Checking...', type: 'loading' });
  
  useEffect(() => {
    const checkStatus = async () => {
      try {
        const res = await fetch('/api/status');
        if (!res.ok) throw new Error('Network response was not ok');
        const data = await res.json();
        setStatus({ 
          text: '🟢 All systems operational', 
          type: 'success' 
        });
      } catch (error) {
        setStatus({ 
          text: '🔴 Some services unavailable', 
          type: 'error' 
        });
      }
    };
    
    checkStatus();
    const interval = setInterval(checkStatus, 30000); // Check every 30 seconds
    return () => clearInterval(interval);
  }, []);
  
  return (
    <div className={`status-indicator ${status.type}`}>
      <div className="status-content">
        <span className="status-text">{status.text}</span>
        <span className="status-timestamp">Last checked: {new Date().toLocaleTimeString()}</span>
      </div>
    </div>
  );
}
